def print_loop(loop_list,indent=False,level=0):
	for loopitem in loop_list:
		if isinstance(loopitem,list):
			print_loop(loopitem,indent,level+1)
		else :
			if indent:
				for tab_stop in range(level):
					print("\t",end='')
			print(loopitem)
