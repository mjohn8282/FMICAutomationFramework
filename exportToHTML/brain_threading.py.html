import astroid


def _thread_transform():
    return astroid.parse(
        """ 
    class lock(object): 
        def acquire(self, blocking=True, timeout=-1): 
            return False 
        def release(self): 
            pass 
        def __enter__(self): 
            return True 
        def __exit__(self, *args): 
            pass 
        def locked(self): 
            return False 
 
    def Lock(): 
        return lock() 
    """
    )


astroid.register_module_extender(astroid.MANAGER, "threading", _thread_transform)
